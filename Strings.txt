======== LOOPING THROUGH A STRING =========

        String fName = "Ibrahim Musa Suleiman";
        int count = 0;
        for(int i=0; i<fName.length(); i++)
        {
            if(fName.charAt(i) != ' ')
            {
                count += 1;
            }
        }
        System.out.println("The number of characters are: " + count);
